        Blueprint.prototype.initialize = function(options){
        var chart = this,
            parentDimensions = this.parent.node().getBoundingClientRect();

        chart.margin = chart.margin || { right: 40, left: 50, top: 20, bottom: 40 };
        chart.w = chart.w || parentDimensions.width - chart.margin.left - chart.margin.right;
        chart.h = chart.h || parentDimensions.height - chart.margin.top - chart.margin.bottom;
